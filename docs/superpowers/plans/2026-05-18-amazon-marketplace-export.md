# Amazon Marketplace Export Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a complete Amazon UAE listing export module that maps product data to the 46-field Amazon feed format, generates listing content (title, bullets, keywords) with AI, validates readiness, and exports a downloadable CSV — with a basic frontend UI to manage and download listings.

**Architecture:** A new `product_marketplace_listings` table stores per-marketplace content (title, description, bullets, keywords) generated by Claude; the `AmazonListingExporter` service assembles all 46 Amazon feed fields from product + listing data; five API endpoints handle CRUD, AI generation, validation, and CSV export. A Next.js frontend page shows validation status per SKU and triggers generation/export.

**Tech Stack:** FastAPI + SQLAlchemy 2.0 async + Alembic + Anthropic SDK (claude-sonnet-4-6) + Pydantic v2 + Next.js 15 + shadcn/ui new-york + TanStack Table + React Query

    from app.db.models.product import Product

MARKETPLACE_VALUES = ("amazon_uae", "noon_uae", "shopify_storefront")

from datetime import datetime
from typing import Any, Literal

from pydantic import BaseModel, ConfigDict, Field

Marketplace = Literal["amazon_uae", "noon_uae", "shopify_storefront"]
ListingStatus = Literal["draft", "ready", "published", "paused"]


class MarketplaceListingRead(BaseModel):
    model_config = ConfigDict(from_attributes=True)

    id: str
    product_sku: str
    marketplace: str
    status: str
    listing_title: str | None
    listing_description: str | None
    bullet_points: list[str]
    search_keywords: str | None
    extra: dict[str, Any]
    ai_generated_at: datetime | None
    ai_model: str | None
    created_at: datetime
    updated_at: datetime


class MarketplaceListingUpsert(BaseModel):
    status: ListingStatus = "draft"
    listing_title: str | None = Field(None, max_length=200)
    listing_description: str | None = Field(None, max_length=2000)
    bullet_points: list[str] = Field(default_factory=list, max_length=5)
    search_keywords: str | None = Field(None, max_length=500)
    extra: dict[str, Any] = Field(default_factory=dict)


class AmazonFieldError(BaseModel):
    field: str
    code: str
    message: str


class AmazonListingValidation(BaseModel):
    sku: str
    is_ready: bool
    errors: list[AmazonFieldError]
    warnings: list[AmazonFieldError]


class AmazonValidationReport(BaseModel):
    total_skus: int
    ready_count: int
    draft_count: int
    error_count: int
    listings: list[AmazonListingValidation]


class GenerateListingRequest(BaseModel):
    overwrite: bool = Field(
        default=False,
        description="If True, regenerate even if content already exists.",
    )

        self,
        product: Any,
        listing: Any | None,
        channel_listing: Any | None,
    ) -> tuple[list[dict], list[dict]]:
        """Return (errors, warnings). Errors block export; warnings are informational."""
        row = self.build_row(product, listing, channel_listing)
        errors: list[dict] = []
        warnings: list[dict] = []

        for field, code, message in _REQUIRED_FIELDS:
            val = row.get(field)
            if not val and val != 0:
                errors.append({"field": field, "code": code, "message": message})

        for field, code, message in _WARNING_RULES:
            val = row.get(field)
            if not val:
                warnings.append({"field": field, "code": code, "message": message})

        bullets_filled = sum(
            1 for k in ("bullet_point1", "bullet_point2", "bullet_point3",
                        "bullet_point4", "bullet_point5")
            if row.get(k)
        )
        if bullets_filled < 5:
            warnings.append({
                "field": "bullet_points",
                "code": "INCOMPLETE_BULLETS",
                "message": f"Only {bullets_filled}/5 bullet points filled. Amazon recommends all 5.",
            })

        return errors, warnings

    def export_csv(
        self,
        rows: list[tuple[Any, Any | None, Any | None]],
    ) -> bytes:
        """Build Amazon flat-file CSV bytes from a list of (product, listing, channel_listing) tuples."""
        buf = io.StringIO()
        writer = csv.DictWriter(
            buf,
            fieldnames=AMAZON_FEED_HEADERS,
            extrasaction="ignore",
            lineterminator="\r\n",
        )
        writer.writeheader()
        for product, listing, channel_listing in rows:
            writer.writerow(self.build_row(product, listing, channel_listing))
        return buf.getvalue().encode("utf-8")

from __future__ import annotations

import logging
from dataclasses import dataclass
from typing import Any

import anthropic

logger = logging.getLogger(__name__)

_GENERATE_TOOL = {
    "name": "save_amazon_listing",
    "description": "Save the generated Amazon listing content.",

            },

                "type": "array",
                "items": {"type": "string"},
                "minItems": 5,
                "maxItems": 5,
                "description": (

                    "4) temperature range, 5) certifications/standards."
                ),
            },
            "search_keywords": {
                "type": "string",
                "description": (
                    "Backend search keywords, comma-separated. "
                    "Include: product type synonyms, material names, DN sizes, standards. "
                    "Max 250 chars."
                ),
            },
        },
        "required": ["listing_title", "listing_description", "bullet_points", "search_keywords"],
    },
}

    "You are an expert Amazon marketplace content specialist for industrial valves and fittings. "
    "You write SEO-optimized product listings for the Amazon UAE marketplace (PlumbingFixture category). "
    "Always write in English. Be technically accurate. Prioritize key specs buyers search for."
)


@dataclass
class GeneratedListingContent:
    listing_title: str
    listing_description: str
    bullet_points: list[str]
    search_keywords: str
    ai_model: str = "claude-sonnet-4-6"


class AmazonListingGenerator:
    """Generates Amazon listing content using Claude tool_use for structured output."""

    def __init__(self, model: str = "claude-sonnet-4-6") -> None:
        self._model = model

    async def generate(self, product_context: dict[str, Any]) -> GeneratedListingContent:
        """Call Claude and return structured listing content.

        Args:
            product_context: Dict with product data keys:
                sku, dn, material, connection_type, pressure_rating,
                temp_min, temp_max, certifications, family, description_en
        """
        user_prompt = _build_user_prompt(product_context)

        client = anthropic.AsyncAnthropic()
        response = await client.messages.create(
            model=self._model,
            max_tokens=1024,
            system=_SYSTEM_PROMPT,
            tools=[_GENERATE_TOOL],
            tool_choice={"type": "tool", "name": "save_amazon_listing"},
            messages=[{"role": "user", "content": user_prompt}],
        )

        if response.stop_reason != "tool_use":
            raise ValueError(
                f"Claude did not return tool_use — stop_reason={response.stop_reason!r}. "
                "Check API key and model availability."
            )

        tool_block = next(b for b in response.content if b.type == "tool_use")
        data = tool_block.input

        return GeneratedListingContent(
            listing_title=data["listing_title"],
            listing_description=data["listing_description"],
            bullet_points=list(data["bullet_points"]),
            search_keywords=data["search_keywords"],
            ai_model=self._model,
        )

        f"Connection type: {ctx.get('connection_type', '')}\n"

        f"Temperature range: {ctx.get('temp_min', '')}°C to {ctx.get('temp_max', '')}°C\n"
        f"Certifications: {certifications}\n"
        f"Technical description: {ctx.get('description_en', '')}\n\n"
        "Use the save_amazon_listing tool to return structured content."
    )

@router.get(
    "/{marketplace}/validate",
    response_model=AmazonValidationReport,
    summary="Validate all active products for marketplace readiness",
    operation_id="marketplaceListingsValidate",
)
async def validate_all(
    marketplace: str,
    session: AsyncSession = Depends(get_db_session),
    user: User = Depends(require_permissions("products:read")),
) -> AmazonValidationReport:
    """Return per-SKU validation result for all active products."""
    if marketplace != "amazon_uae":
        raise HTTPException(status_code=400, detail=f"Unsupported marketplace: {marketplace!r}")

    products = await _load_all_active_products(session)
    listings: list[AmazonListingValidation] = []
    ready_count = 0
    error_count = 0
    draft_count = 0

    for product in products:
        mp_listing = next(
            (ml for ml in product.marketplace_listings if ml.marketplace == marketplace), None
        )
        channel_listing = await _get_channel_listing(product.sku, marketplace, session)
        errors_raw, warnings_raw = _EXPORTER.validate(product, mp_listing, channel_listing)

        is_ready = len(errors_raw) == 0
        if is_ready:
            ready_count += 1
        else:
            error_count += 1
        if mp_listing is None or mp_listing.status == "draft":
            draft_count += 1

        listings.append(AmazonListingValidation(
            sku=product.sku,
            is_ready=is_ready,
            errors=[AmazonFieldError(**e) for e in errors_raw],
            warnings=[AmazonFieldError(**w) for w in warnings_raw],
        ))

    return AmazonValidationReport(
        total_skus=len(products),
        ready_count=ready_count,
        draft_count=draft_count,
        error_count=error_count,
        listings=listings,
    )

@router.post(
    "/{sku}/{marketplace}/generate",
    response_model=MarketplaceListingRead,
    summary="AI-generate listing content from product data",
    operation_id="marketplaceListingsGenerate",
)
async def generate_listing(
    sku: str,
    marketplace: str,
    body: GenerateListingRequest,
    session: AsyncSession = Depends(get_db_session),
    user: User = Depends(require_permissions("products:write")),
) -> MarketplaceListing:
    """Call Claude to generate listing_title, listing_description, bullets, keywords.

    Skips generation if content already exists unless `overwrite=true`.
    """
    product = await _get_product_or_404(sku, session)

    result = await session.execute(
        select(MarketplaceListing)
        .where(MarketplaceListing.product_sku == sku, MarketplaceListing.marketplace == marketplace)
    )
    listing = result.scalar_one_or_none()

    if listing and listing.listing_title and not body.overwrite:
        raise HTTPException(
            status_code=status.HTTP_409_CONFLICT,
            detail=f"Listing already has content. Pass overwrite=true to regenerate.",
        )

    # Build product context for the generator
    body_material = next(
        (m.material for m in product.materials if m.component == "body" and m.position == 0),
        getattr(product, "material", ""),
    )
    first_conn = next(
        (c.connection_type for c in sorted(product.connections, key=lambda c: c.position)),
        "",
    )
    pt = next(
        (t for t in product.tech_tables if t.kind == "pressure_temperature"), None
    )

    translations = getattr(product, "translations", []) or []
    en_translation = next((t for t in translations if t.lang == "en"), None)
    description_en = getattr(en_translation, "description", "") or ""

    context = {
        "sku": product.sku,
        "dn": getattr(product, "dn", ""),
        "material": body_material,
        "connection_type": first_conn,
        "pressure_rating": pt.data.get("pn") if pt else "",
        "temp_min": pt.data.get("temp_min_c") if pt else "",
        "temp_max": pt.data.get("temp_max_c") if pt else "",
        "certifications": [c.certification_code for c in product.certificates],
        "family": getattr(product, "family", ""),
        "description_en": description_en,
    }

    try:
        generated = await _GENERATOR.generate(context)
    except Exception as exc:
        logger.exception("AI listing generation failed for SKU %s: %s", sku, exc)
        raise HTTPException(status_code=500, detail=f"AI generation failed: {exc}") from exc

    now = datetime.now(tz=timezone.utc)

    if listing is None:
        listing = MarketplaceListing(product_sku=sku, marketplace=marketplace)
        session.add(listing)

    listing.listing_title = generated.listing_title
    listing.listing_description = generated.listing_description
    listing.bullet_points = generated.bullet_points
    listing.search_keywords = generated.search_keywords
    listing.ai_generated_at = now
    listing.ai_model = generated.ai_model
    listing.status = "draft"
    listing.updated_at = now

    await session.flush()
    await session.refresh(listing)
    return listing

const API_BASE = process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:8000";

async function apiFetch<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${API_BASE}/api/v1${path}`, {
    ...init,
    headers: {
      "Content-Type": "application/json",
      ...(init?.headers ?? {}),
    },
    credentials: "include",
  });
  if (!res.ok) {
    const text = await res.text().catch(() => res.statusText);
    throw new Error(`API ${res.status}: ${text}`);
  }
  return res.json() as Promise<T>;
}

export async function fetchAmazonValidation(): Promise<AmazonValidationReport> {
